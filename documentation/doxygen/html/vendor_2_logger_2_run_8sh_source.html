76## @fn tcli_linux_bash_logger_infoscreenFailed()
77## @details
78## **Info of the process step [ FAILED ]**
79## @param string error message part 1 red color
80## @param string error message part 2 blue color
81## @param string error message part 3 red color
82## @param string function or other notice in front of message
83tcli_linux_bash_logger_infoscreenFailed() {
84 local -a _errormsg
85 [ ${1:-} ] && _errormsg=($1)
86 [ ${2:-} ] && _errormsg+=($2)
87 [ ${3:-} ] && _errormsg+=($3)
88 [ ${TCLI_LINUX_BASH_LOGGER_INFOSCREEN_WARN} == 1 ] && TCLI_LINUX_BASH_LOGGER_INFOSCREEN_WARN=0
89 printf "${TCLI_LINUX_BASH_BACK_TO_LINE}${TCLI_LINUX_BASH_LOGGER_RED}FAILED${TCLI_LINUX_BASH_LOGGER_NC}\n"
90 [ ${TCLI_LINUX_BASH_LOGGER_LOT_TO_FILE} == 1 ] && tcli_linux_bash_logger_file_error $(echo ${_errormsg[@]}) ${4:-}
91}

93## @fn tcli_linux_bash_logger_infoscreenFailedExit()
94## @details
95## **Info of the process step [ FAILED ]**
96## Then it will exit with a error code
97## @param string error message part 1 red color
98## @param string error message part 2 blue color
99## @param string error message part 3 red color
100## @param interger exit code (default is 1)
101## @param string error cause
102tcli_linux_bash_logger_infoscreenFailedExit() {
103 tcli_linux_bash_logger_infoscreenFailed "${1:-}" "${2:-}" "${3:-}" "${5:-}"
104 [ ${4:-} ] && printf "${TCLI_LINUX_BASH_LOGGER_RED}${1} : "
105 [ ${1:-} ] && printf "${TCLI_LINUX_BASH_LOGGER_RED}${1}"
106 [ ${2:-} ] && printf " ${TCLI_LINUX_BASH_LOGGER_BLUE}${2}"
107 [ ${3:-} ] && printf " ${TCLI_LINUX_BASH_LOGGER_RED}${3}"
108 printf "${TCLI_LINUX_BASH_LOGGER_NC}\n"
109 exit ${4:-1}
110}

143## @fn tcli_linux_bash_logger_title()
144## @details
145## **Title**
146## Creating a box around title
147## Fillerlength is numbers of char horizontialt
148## @param string title
149## @param string fillerlength ( default value 80 )
150## @param char filler char ( default value +)
151tcli_linux_bash_logger_title() {
152 local _title=${1:-}
153 local -i _fillerlength=${2:-80}
154 ((_fillerlength--))
155 local _fillerchar=${3:-"+"}
156 local -i _n_pad=$(((${_fillerlength} - ${#_title} - 2) / 2))
157 local -i _i=$(($_n_pad * 2 + 2 + ${#_title}))
158 printf -- "${_fillerchar}%.0s" $(seq ${_fillerlength})
159 printf "\n"
160 printf -- "${_fillerchar}%.0s" $(seq ${_n_pad})
161 printf $(printf ' %s ' "$_title")
162 if [ $(($_n_pad * 2 + ${#_title} + 2)) -lt ${_fillerlength} ]; then
163 ((_n_pad++))
164 fi
165 printf -- "${_fillerchar}%.0s" $(seq ${_n_pad})
166 printf "\n"
167 printf -- "${_fillerchar}%.0s" $(seq ${_fillerlength})
168 printf "\n"
169}

198## @fn tcli_linux_bash_logger_file()
199## @details
200## **Error to log file**
201## @param string level message
202## @param string message
203## @param string function or other notice before message
204tcli_linux_bash_logger_file() {
205 local _msg
206 if [[ -n "${2:-}" && -n "${3:-}" ]]; then
207 _msg="${3} : ${2}"
208 elif [[ -n "${2:-}" ]]; then
209 _msg="${2}"
210 elif [[ -n "${3:-}" ]]; then
211 _msg="${3}"
212 else
213 tcli_linux_bash_logger_file_warn "tcli_linux_bash_logger_file" "missing message"
214 return 1
215 fi
216 echo "[$(date +%Y-%m-%d\ %T.%6N)] ${1:-} >>> ${_msg}" >&2
217}
