32# This script determines the Linux distribution and its version.
33# It sets the following environment variables:
34# - TCLI_LINUX_BASH_DISTRIBUTION_ID: The name of the Linux distribution.
35# - TCLI_LINUX_BASH_DISTRIBUTION_RELEASE: The version of the Linux distribution.
36#
37# The script checks for the distribution information in the following order:
38# 1. /etc/os-release: Standard file for freedesktop.org and systemd-based systems.
39# 2. lsb_release command: Standard command for Linux Standard Base (LSB) compliant distributions.
40# 3. /etc/lsb-release: Fallback for some versions of Debian/Ubuntu without the lsb_release command.
41# 4. /etc/debian_version: Fallback for older Debian/Ubuntu systems.
42# 5. uname command: Fallback for other systems, including BSD.
43if [ -f /etc/os-release ]; then
44 # freedesktop.org and systemd
45 . /etc/os-release
46 TCLI_LINUX_BASH_DISTRIBUTION_ID=$NAME
47 TCLI_LINUX_BASH_DISTRIBUTION_RELEASE=$VERSION_ID
48elif type lsb_release >/dev/null 2>&1; then
49 # linuxbase.org
50 TCLI_LINUX_BASH_DISTRIBUTION_ID=$(lsb_release -si)
51 TCLI_LINUX_BASH_DISTRIBUTION_RELEASE=$(lsb_release -sr)
52elif [ -f /etc/lsb-release ]; then
53 # For some versions of Debian/Ubuntu without lsb_release command
54 . /etc/lsb-release
55 TCLI_LINUX_BASH_DISTRIBUTION_ID=$DISTRIB_ID
56 TCLI_LINUX_BASH_DISTRIBUTION_RELEASE=$DISTRIB_RELEASE
57elif [ -f /etc/debian_version ]; then
58 # Older Debian/Ubuntu/etc.
59 TCLI_LINUX_BASH_DISTRIBUTION_ID=Debian
60 TCLI_LINUX_BASH_DISTRIBUTION_RELEASE=$(cat /etc/debian_version)
61else
62 # Fall back to uname, e.g. "Linux <version>", also works for BSD, etc.
63 TCLI_LINUX_BASH_DISTRIBUTION_ID=$(uname -s)
64 TCLI_LINUX_BASH_DISTRIBUTION_RELEASE=$(uname -r)
65fi
66
67# This script determines the architecture of the current machine using the `uname -m` command
68# and sets the TCLI_LINUX_BASH_DISTRIBUTION_ARCH variable accordingly.
69# Supported architectures:
70# - x86_64: AMD64 or Intel64
71# - i*86: IA32 or Intel32
72# - arm*: ARM
73# - riscv64: RISC-V 64-bit
74# If the architecture is not recognized, it defaults to the output of `uname -m`.
75case $(uname -m) in
76x86_64)
77 TCLI_LINUX_BASH_DISTRIBUTION_ARCH=x64 # AMD64 or Intel64
78 ;;
79i*86)
80 TCLI_LINUX_BASH_DISTRIBUTION_ARCH=x86 # IA32 or Intel32
81 ;;
82arm*)
83 TCLI_LINUX_BASH_DISTRIBUTION_ARCH=arm # ARM
84 ;;
85riscv64)
86 TCLI_LINUX_BASH_DISTRIBUTION_ARCH=riscv64 # RISC-V 64-bit
87 ;;
88*)
89 TCLI_LINUX_BASH_DISTRIBUTION_ARCH=$(uname -m)
90 ;;
91esac
