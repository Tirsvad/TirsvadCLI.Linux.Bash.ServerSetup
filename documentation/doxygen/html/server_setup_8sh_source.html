224## @fn can_connect_server()
225## @brief Check if a server can be connected to
226## @details
227## This function checks if a server can be connected to
228## @param server The server host
229## @param port The server port
230## @param password The root password
231## @return 1 if the server can not be connected to
232## @return 2 if the server can not be connected to via ssh
233can_connect_server() {
234 local server=$1
235 local port=$2
236 local password=$3
237 local port_hardness=$4
238
239 nc -z -w 2 $server $port > /dev/null
240 if [ $? -ne 0 ]; then
241 nc -z -w 2 $server $port_hardness > /dev/null
242 if [ $? -ne 0 ]; then
243 return 1
244 fi
245 SERVER_PORT=$SERVER_PORT_HARDNESS
246 sshpass -p $password ssh -o StrictHostKeyChecking=no -o ConnectTimeout=4 root@$server -p $port_hardness exit > /dev/null
247 if [ $? -ne 0 ]; then
248 return 2
249 fi
250 else
251 sshpass -p $password ssh -o StrictHostKeyChecking=no -o ConnectTimeout=4 root@$server -p $port exit > /dev/null
252 if [ $? -ne 0 ]; then
253 return 2
254 fi
255 fi
256}
