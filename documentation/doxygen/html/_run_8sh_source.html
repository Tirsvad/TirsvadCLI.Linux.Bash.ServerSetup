65## @fn check_server_connection_handler()
66## @brief Check server connection
67## @details
68## This function checks if the server can be connected
69## @exit 1 if the server can not be connected
70## @exit 2 if the SSH can not be connected
71check_server_connection_handler() {
72 tcli_linux_bash_logger_infoscreen "Checking Server Connection"
73 can_connect_server $SERVER_HOST $SERVER_PORT $ROOT_PASSWORD || {
74 err=$?
75 tcli_linux_bash_logger_infoscreenFailed
76 if [ $err -eq 1 ]; then
77 printf "\n${RED}Could not connect to the server ${SERVER_HOST}:${SERVER_PORT} ${NC}\n"
78 exit $err
79 fi
80 if [ $err -eq 2 ]; then
81 printf "\n${RED}SSH could not be connected{NC}\n"
82 exit $err
83 fi
84 printf "\n${RED}Could not connect to the server. UNKNOW why!${NC}\n"
85 exit $err
86 }
87 tcli_linux_bash_logger_infoscreenDone
88}
