65## @fn check_server_connection_handler()
66## @brief Check server connection
67## @details
68## This function checks if the server can be connected
69## @exit 1 if the server can not be connected
70## @exit 2 if the SSH can not be connected
71check_server_connection_handler() {
72 tcli_linux_bash_logger_infoscreen "Checking Server Connection"
73 can_connect_server $SERVER_HOST $SERVER_PORT $ROOT_PASSWORD $SERVER_PORT_HARDNESS || {
74 err=$?
75 if [ $err -eq 1 ]; then
76 tcli_linux_bash_logger_infoscreenFailed
77 printf "\n${RED}Could not connect to the server ${SERVER_HOST}:${SERVER_PORT} ${NC}\n"
78 exit $err
79 fi
80 if [ $err -eq 2 ]; then
81 tcli_linux_bash_logger_infoscreenWarn
82 tcli_linux_bash_logger_file_warn "if we already have a user with sudo rights, we can ignore this warning" "SSH could not be connected with root user"
83 return 0
84 fi
85 printf "\n${RED}Could not connect to the server. UNKNOW error!${NC}\n"
86 exit $err
87 }
88 tcli_linux_bash_logger_infoscreenDone
89}
